globalFolderSortable = null; } globalItemSortables.forEach(s => { if(s) s.destroy(); }); globalItemSortables = []; currentSortMode = 0; const btn = document.getElementById('globalSortMasterBtn'); if(btn) { btn.className = 'toggle-sort-btn'; btn.innerText = '↕️ 정렬 모드: 닫힘'; } } /* [개선] 버튼을 누르면 파일과 내부 물품 정렬 기능이 동시에 켜지는 함수 */ function rotateSortMode() { currentSortMode = (currentSortMode === 0) ? 1 : 0; const btn = document.getElementById('globalSortMasterBtn'); const container = document.getElementById('itemStatusList'); // 기존 인스턴스 및 클래스 전면 초기화 if(globalFolderSortable) { globalFolderSortable.destroy(); globalFolderSortable = null; } globalItemSortables.forEach(s => { if(s) s.destroy(); }); globalItemSortables = []; document.querySelectorAll('.category-section').forEach(s => s.classList.remove('sort-mode-folder-active')); document.querySelectorAll('.item-card').forEach(c => c.classList.remove('sort-mode-item-active')); btn.className = 'toggle-sort-btn'; if (currentSortMode === 0) { // 정렬 종료 및 저장 btn.innerText = '↕️ 정렬 모드: 닫힘'; alert('🔒 모든 대분류 파일 및 내부 물품의 배치가 안전하게 저장되었습니다.'); loadInventoryStatus(); } else { // 🔄 통합 편집 및 정렬 모드 활성화 (둘 다 동시 켜짐) btn.innerText = '🔄 통합 편집 및 정렬 모드'; btn.classList.add('folder-mode'); // 기존 스타일 활용 (파란색 버튼) // 1. 대분류 파일(폴더) 점선 및 드래그 활성화 document.querySelectorAll('.category-section').forEach(s => s.classList.add('sort-mode-folder-active')); globalFolderSortable = Sortable.create(container, { animation: 200, ghostClass: 'sortable-ghost', chosenClass: 'sortable-chosen', onEnd() { const sections = container.querySelectorAll('.category-section'); for(let i = 0; i < cards.length; i++) { const id = cards[i].getAttribute('data-itemid'); await fetch(`${BASE_URL}/items?id=eq.${id}`, { method: 'PATCH', headers, body: JSON.stringify({ sort_order: i + 1 }) }); } } }); globalItemSortables.push(sortableInstance); }); } }
