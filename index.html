import openpyxl from openpyxl.styles import Font, PatternFill, Alignment, Border, Side from openpyxl.utils import get_column_letter # 1. 워크북 및 시트 생성 wb = openpyxl.Workbook() ws = wb.active ws.title = "자재관리" ws.views.sheetView[0].showGridLines = True # 눈금선 표시 # 2. 스타일 정의 font_title = Font(name="맑은 고딕", size=14, bold=True) font_header = Font(name="맑은 고딕", size=11, bold=True) font_data = Font(name="맑은 고딕", size=11) fill_yellow = PatternFill(start_color="FFFF00", end_color="FFFF00", fill_type="solid") fill_gray = PatternFill(start_color="D9D9D9", end_color="D9D9D9", fill_type="solid") align_center = Alignment(horizontal="center", vertical="center") align_left = Alignment(horizontal="left", vertical="center") align_right = Alignment(horizontal="right", vertical="center") thin_border = Border( left=Side(style="thin", color="D3D3D3"), right=Side(style="thin", color="D3D3D3"), top=Side(style="thin", color="D3D3D3"), bottom=Side(style="thin", color="D3D3D3") ) # 3. 데이터 정의 # (1) 제품 총 재고 현황 stock_title = "2026년 7월 제품 총 재고 현황" stock_headers = ["분류", "제품명", "1일 총 개수", "마지막일 총 개수"] stock_data = [ ["조명제어", "1구", 0, 222], ["피크제어", "리얼시스", 0, 10], ["피크제어", "삼성중계기", 0, 123] ] # (2) 입고 현황 receiving_title = "입 고 현 황" receiving_headers = ["입고날짜", "분류", "제품명", "입고수량", "비고"] receiving_data = [ ["2026-07-06", "피크제어", "리얼시스", 5, ""], ["2026-07-06", "기존분류", "ㄴㅇㄹ", 123, "asd"], ["2026-07-06", "피크제어", "리얼시스", 8, ""], ["2026-07-06", "피크제어", "삼성중계기", 123, ""], ["2026-07-06", "조명제어", "1구", 222, "DF"] ] # (3) 출고 현황 shipping_title = "출 고 현 황" shipping_headers = ["출고날짜", "분류", "제품명", "출고수량", "비고"] shipping_data = [ ["2026-07-06", "피크제어", "리얼시스", 3, "예시데이터"] # 필요에 따라 변경 가능 ] # 4. 엑셀에 데이터 및 서식 작성 함수 def write_table(start_row, title, headers, data, is_stock_table=False): # 제목 작성 ws.cell(row=start_row, column=1, value=title) if is_stock_table: # '제품 총 재고 현황'만 A열부터 D열까지 노란색 채우기 for col in range(1, 5): cell = ws.cell(row=start_row, column=col) cell.fill = fill_yellow cell.font = font_title else: # 입고/출고 현황은 텍스트만 서식 적용 (배경색 없음) ws.cell(row=start_row, column=1).font = font_title # 헤더 작성 header_row = start_row + 2 for col_idx, header in enumerate(headers, 1): cell = ws.cell(row=header_row, column=col_idx, value=header) cell.font = font_header cell.fill = fill_gray cell.alignment = align_left if col_idx <= 2 else align_center cell.border = thin_border # 데이터 작성 current_row = header_row + 1 for row_data in data: for col_idx, value in enumerate(row_data, 1): cell = ws.cell(row=current_row, column=col_idx, value=value) cell.font = font_data cell.border = thin_border # 열 종류별 정렬 및 서식 설정 if is_stock_table: if col_idx in [1, 2]: # 분류, 제품명 cell.alignment = align_left else: # 개수 숫자는 오른쪽 정렬 및 천단위 콤마 cell.alignment = align_right cell.number_format = '#,##0' else: if col_idx == 1: # 날짜 cell.alignment = align_center elif col_idx in [2, 3]: # 분류, 제품명 cell.alignment = align_left elif col_idx == 4: # 수량 cell.alignment = align_right cell.number_format = '#,##0' else: # 비고 cell.alignment = align_left current_row += 1 return current_row + 2 # 다음 표를 위한 시작 행 반환 (두 줄 띄움) # 5. 각 표 그리기 next_row = write_table(2, stock_title, stock_headers, stock_data, is_stock_table=True) next_row = write_table(next_row, receiving_title, receiving_headers, receiving_data) next_row = write_table(next_row, shipping_title, shipping_headers, shipping_data) # 6. 열 너비 자동 조절 for col in ws.columns: max_len = 0 col_letter = get_column_letter(col[0].column) for cell in col: if cell.value: # 한글 글자 깨짐 방지를 위해 인코딩 길이 고려 val_str = str(cell.value) cell_len = len(val_str.encode('utf-8')) if cell_len > max_len: max_len = cell_len # 제목 줄 때문에 너무 넓어지는 것 방지 (최대 너비 제한 및 최소 너비 보장) ws.column_dimensions[col_letter].width = min(max(max_len // 2 + 3, 12), 25) # 7. 파일 저장 (요청하신 파일명 반영) wb.save("시스템사업부 자재관리.xlsx") print("파일 생성 완료: 시스템사업부 자재관리.xlsx")
